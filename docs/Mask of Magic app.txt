import React, { useState, useCallback, useRef, useEffect } from 'react';
import { Editor } from './components/Editor';
import { Controls } from './components/Controls';
import { DebugView } from './components/DebugView';
import { AppState, FloatingObject } from './types';
import { Flow } from 'flow-sdk';
import { createOneStepComposite } from './utils/imageUtils';

export default function App() {
  const [state, setState] = useState<AppState>({
    originalImage: null,
    history: [],
    historyIndex: -1,
    maskCanvas: null,
    tool: 'brush', // Chỉ giữ lại cọ vẽ
    editMode: 'in-place',
    moveSubMode: 'one-step',
    floatingObject: null,
    brushSize: 12,
    prompt: '',
    isGenerating: false,
    status: 'Đang chờ ảnh...',
    modelVariant: 'nano-banana-pro',
    highQuality: true,
    debugMode: false,
    debugSteps: []
  });

  const [contextMenuPos, setContextMenuPos] = useState<{ x: number, y: number } | null>(null);
  const contextMenuInputRef = useRef<HTMLInputElement>(null);

  const handleUndo = useCallback(() => {
    setState(prev => {
      if (prev.historyIndex > 0) {
        const newIndex = prev.historyIndex - 1;
        return {
          ...prev,
          historyIndex: newIndex,
          originalImage: prev.history[newIndex],
          floatingObject: null,
          maskCanvas: null
        };
      }
      return prev;
    });
  }, []);

  const handleRedo = useCallback(() => {
    setState(prev => {
      if (prev.historyIndex < prev.history.length - 1) {
        const newIndex = prev.historyIndex + 1;
        return {
          ...prev,
          historyIndex: newIndex,
          originalImage: prev.history[newIndex],
          floatingObject: null,
          maskCanvas: null
        };
      }
      return prev;
    });
  }, []);

  useEffect(() => {
    const handleGlobalKeyDown = (e: KeyboardEvent) => {
      const target = e.target as HTMLElement;
      const isInput = target.tagName === 'INPUT' || target.tagName === 'TEXTAREA';

      if ((e.ctrlKey || e.metaKey) && !isInput) {
        if (e.key === 'z') {
          e.preventDefault();
          if (e.shiftKey) {
            handleRedo();
          } else {
            handleUndo();
          }
          return;
        }
        if (e.key === 'y') {
          e.preventDefault();
          handleRedo();
          return;
        }
      }

      if (e.key === 'Enter' && !e.shiftKey) {
        if (isInput) {
          if (target !== contextMenuInputRef.current) return;
        }
        
        if (!state.isGenerating && state.originalImage) {
          if (state.editMode === 'in-place' && state.prompt) {
            handleGenerate();
            setContextMenuPos(null);
          } else if (state.editMode === 'move' && state.floatingObject) {
            handleGenerate();
            setContextMenuPos(null);
          }
        }
      }
    };
    window.addEventListener('keydown', handleGlobalKeyDown);
    return () => window.removeEventListener('keydown', handleGlobalKeyDown);
  }, [state.isGenerating, state.originalImage, state.editMode, state.prompt, state.floatingObject, handleUndo, handleRedo]);

  const handleSelectImage = async () => {
    setState(prev => ({ ...prev, status: 'Đang mở thư viện...' }));
    try {
      const media = await Flow.media.select({ filter: 'image' });
      const dataUrl = `data:${media.mimeType};base64,${media.base64}`;
      const img = new Image();
      img.src = dataUrl;
      img.onload = () => {
        const newImageState = {
          base64: media.base64,
          dataUrl,
          width: img.width,
          height: img.height,
          mimeType: media.mimeType
        };
        setState(prev => ({
          ...prev,
          originalImage: newImageState,
          history: [newImageState],
          historyIndex: 0,
          status: 'Đã tải ảnh',
          floatingObject: null,
          maskCanvas: null
        }));
      };
    } catch (err) {
      console.error('Lỗi chọn ảnh:', err);
      setState(prev => ({ ...prev, status: 'Đã hủy chọn' }));
    }
  };

  const handleMaskUpdate = useCallback((canvas: HTMLCanvasElement | null) => {
    setState(prev => ({ ...prev, maskCanvas: canvas }));
  }, []);

  const handleSetFloatingObject = useCallback((obj: FloatingObject | null) => {
    setState(prev => ({ ...prev, floatingObject: obj }));
  }, []);

  const uploadToFlow = async (base64: string, name: string) => {
    const result = await Flow.upload({
      base64,
      mimeType: 'image/png',
      name
    });
    return result.mediaId;
  };

  const callGenerateImage = async (promptText: string, imageBase64: string, maskBase64?: string) => {
    const imageMediaId = await uploadToFlow(imageBase64, 'Ảnh nguồn');
    const referenceIds = [imageMediaId];
    
    if (maskBase64) {
      const maskMediaId = await uploadToFlow(maskBase64, 'Mặt nạ');
      referenceIds.push(maskMediaId);
    }

    const modelDisplayName = state.modelVariant === 'nano-banana-2' ? '🍌 Nano Banana 2' : '🍌 Nano Banana Pro';

    const width = state.originalImage!.width;
    const height = state.originalImage!.height;
    const ratio = width / height;
    
    const targets = [
      { name: '1:1', val: 1 },
      { name: '16:9', val: 16 / 9 },
      { name: '9:16', val: 9 / 16 },
      { name: '4:3', val: 4 / 3 },
      { name: '3:4', val: 3 / 4 },
    ] as const;
    
    const closest = targets.reduce((prev, curr) => 
      Math.abs(curr.val - ratio) < Math.abs(prev.val - ratio) ? curr : prev
    );

    const qualitySuffix = state.highQuality 
      ? ", 8k resolution, highly detailed, photorealistic, sharp focus, masterpiece, cinematic lighting" 
      : "";

    const finalPrompt = `${promptText}. Match the exact aspect ratio and framing of the reference image${qualitySuffix}.`;

    const result = await Flow.generate.image({
      prompt: finalPrompt,
      referenceImageMediaIds: referenceIds,
      modelDisplayName,
      aspectRatio: closest.name as any
    });

    return result.base64;
  };

  const updateImage = (base64: string) => {
    const dataUrl = `data:image/png;base64,${base64}`;
    const img = new Image();
    img.src = dataUrl;
    img.onload = () => {
      setState(prev => {
        const newImageState = {
          base64,
          dataUrl,
          width: img.width,
          height: img.height,
          mimeType: 'image/png'
        };

        const newHistory = prev.history.slice(0, prev.historyIndex + 1);
        newHistory.push(newImageState);

        return {
          ...prev,
          originalImage: newImageState,
          history: newHistory,
          historyIndex: newHistory.length - 1,
          isGenerating: false,
          status: 'Hoàn tất!',
          floatingObject: null,
          maskCanvas: null
        };
      });
    };
  };

  const handleGenerate = async () => {
    if (!state.originalImage || state.isGenerating) return;
    if (state.editMode === 'in-place' && !state.prompt) return;

    setState(prev => ({ ...prev, isGenerating: true, status: 'Đang xử lý...', debugSteps: [], prompt: '' }));
    
    try {
      if (state.editMode === 'in-place') {
        let maskBase64 = '';
        if (state.maskCanvas) {
          const tempCanvas = document.createElement('canvas');
          tempCanvas.width = state.maskCanvas.width;
          tempCanvas.height = state.maskCanvas.height;
          const ctx = tempCanvas.getContext('2d')!;
          ctx.drawImage(state.maskCanvas, 0, 0);
          
          const imgData = ctx.getImageData(0, 0, tempCanvas.width, tempCanvas.height);
          for (let i = 3; i < imgData.data.length; i += 4) {
            if (imgData.data[i] > 0) {
              imgData.data[i] = 255;
              imgData.data[i-1] = 255;
              imgData.data[i-2] = 255;
              imgData.data[i-3] = 255;
            }
          }
          ctx.putImageData(imgData, 0, 0);
          maskBase64 = tempCanvas.toDataURL('image/png').split(',')[1];
        }

        const promptText = `Nhiệm vụ: ${state.prompt}. Hòa quyện tự nhiên với nền. Mặt nạ chỉ định vùng cần sửa.`;
        const resultBase64 = await callGenerateImage(promptText, state.originalImage.base64, maskBase64);

        if (state.debugMode) {
          setState(prev => ({
            ...prev,
            debugSteps: [{
              phase: 'Chỉnh sửa tại chỗ',
              prompt: promptText,
              image: state.originalImage!.base64,
              mask: maskBase64,
              result: resultBase64
            }]
          }));
        }
        
        updateImage(resultBase64);
      } else if (state.editMode === 'move' && state.floatingObject) {
        const origImg = new Image();
        origImg.src = state.originalImage.dataUrl;
        await new Promise(resolve => { origImg.onload = resolve; });
        const origCanvas = document.createElement('canvas');
        origCanvas.width = origImg.width;
        origCanvas.height = origImg.height;
        origCanvas.getContext('2d')!.drawImage(origImg, 0, 0);

        const { imageBase64, maskBase64 } = createOneStepComposite(origCanvas, state.floatingObject);
        const promptText = `Lấp đầy khoảng trống và hòa hợp vị trí mới của đối tượng. Giữ nguyên phong cách và ánh sáng.`;
        const resultBase64 = await callGenerateImage(promptText, imageBase64, maskBase64);
        
        updateImage(resultBase64);
      }
    } catch (err: any) {
      console.error('Lỗi xử lý:', err);
      setState(prev => ({ 
        ...prev, 
        isGenerating: false, 
        status: `Lỗi: ${err.message || 'kiểm tra kết nối'}` 
      }));
    }
  };

  const handleResetMask = () => {
    if (state.maskCanvas) {
      const ctx = state.maskCanvas.getContext('2d');
      if (ctx) ctx.clearRect(0, 0, state.maskCanvas.width, state.maskCanvas.height);
    }
    setState(prev => ({ ...prev, floatingObject: null, status: 'Đã xóa mặt nạ' }));
  };

  const handleSave = async () => {
    if (!state.originalImage) return;
    try {
      await Flow.save({
        base64: state.originalImage.base64,
        mimeType: 'image/png',
        name: `Banana_Edit_${Date.now()}`
      });
      setState(prev => ({ ...prev, status: 'Đã lưu vào Thư viện' }));
    } catch (err) {
      console.error('Lưu thất bại:', err);
    }
  };

  const handleDone = () => {
    setState(prev => {
      const baseHistory = prev.history.length > 0 ? [prev.history[0]] : [];
      return {
        ...prev,
        originalImage: prev.history[0] || null,
        history: baseHistory,
        historyIndex: baseHistory.length > 0 ? 0 : -1,
        maskCanvas: null,
        tool: 'brush', // Giữ cọ vẽ
        floatingObject: null,
        prompt: '',
        status: 'Đã sẵn sàng',
        debugSteps: []
      };
    });
  };

  return (
    <div className="fixed inset-0 flex flex-row bg-white font-mono text-black overflow-hidden">
      <main className="flex-1 relative flex flex-col min-h-0 min-w-0 bg-checkerboard">
        {!state.originalImage ? (
          <div className="flex-1 flex flex-col items-center justify-center gap-6 p-8">
            <div className="border-[3px] border-black bg-white p-6 md:p-10 flex flex-col items-center gap-6 max-w-[420px] w-full text-center shadow-[6px_6px_0px_#000]">
              <div className="space-y-3 w-full text-left">
                <span className="flex items-center justify-between pb-2 mb-2 w-full font-black text-[22px] tracking-widest uppercase">
                  <span>HƯỚNG DẪN</span>
                  <span className="h-[4px] bg-black flex-1 ml-4 block mt-1"></span>
                </span>
                <span className="block border-[3px] border-black p-3 bg-white font-bold text-lg uppercase">1. CHỌN ẢNH TỪ MÁY</span>
                <span className="block border-[3px] border-black p-3 bg-white font-bold text-lg uppercase">2. TÔ VÙNG CẦN SỬA</span>
                <span className="block border-[3px] border-black p-3 bg-white font-bold text-lg uppercase">3. NHẬP LỆNH CHỈNH SỬA</span>
              </div>
              <button
                onClick={handleSelectImage}
                className="w-full px-8 py-3.5 border-[3px] border-black bg-black text-white font-bold text-xl tracking-widest uppercase hover:bg-white hover:text-black transition-none mt-2 shadow-[2px_2px_0px_#000] active:translate-y-1 active:translate-x-1 active:shadow-none"
              >
                MỞ THƯ VIỆN ẢNH
              </button>
            </div>
          </div>
        ) : (
          <Editor
            image={state.originalImage.dataUrl}
            tool={state.tool}
            brushSize={state.brushSize}
            onMaskUpdate={handleMaskUpdate}
            status={state.status}
            setStatus={(s) => setState(prev => ({ ...prev, status: s }))}
            editMode={state.editMode}
            floatingObject={state.floatingObject}
            setFloatingObject={handleSetFloatingObject}
            isGenerating={state.isGenerating}
            onPromptRequest={(pos) => {
              if (!state.isGenerating && state.originalImage) {
                setContextMenuPos(pos);
                setTimeout(() => contextMenuInputRef.current?.focus(), 50);
              }
            }}
          />
        )}

        {contextMenuPos && (
          <>
            <div className="fixed inset-0 z-40" onClick={() => setContextMenuPos(null)} />
            <div 
              className="fixed z-50 bg-white border-[3px] border-black p-2 flex gap-2 items-center shadow-[4px_4px_0px_#000]"
              style={{ left: contextMenuPos.x, top: contextMenuPos.y }}
            >
              <input
                ref={contextMenuInputRef}
                type="text"
                placeholder={state.editMode === 'move' ? "THỰC THI DI CHUYỂN..." : "NHẬP LỆNH CHỈNH SỬA..."}
                value={state.prompt}
                onChange={(e) => setState(prev => ({ ...prev, prompt: e.target.value }))}
                className="bg-transparent border-[3px] border-black text-black placeholder:text-gray-500 font-bold px-3 py-2 w-72 uppercase text-lg outline-none focus:bg-[#f4f4f4]"
                onKeyDown={(e) => {
                  if (e.key === 'Enter') {
                    e.preventDefault();
                    handleGenerate();
                    setContextMenuPos(null);
                  }
                }}
              />
              <button
                onClick={() => {
                  handleGenerate();
                  setContextMenuPos(null);
                }}
                className="bg-white text-black border-[3px] border-black px-4 py-2 font-bold hover:bg-black hover:text-white transition-none uppercase text-lg"
              >
                CHẠY
              </button>
            </div>
          </>
        )}

        {state.originalImage && state.history.length > 1 && (
          <div className="absolute bottom-6 left-1/2 -translate-x-1/2 flex items-center bg-white border-[3px] border-black p-2 pointer-events-auto shadow-[4px_4px_0px_#000] gap-2 overflow-x-auto max-w-[80vw]">
            <div className="flex flex-col items-center justify-center pr-4 border-r-[3px] border-black mr-2 shrink-0">
               <span className="font-bold text-xs uppercase tracking-widest leading-none mb-1">LỊCH SỬ</span>
               <span className="font-bold text-[10px] text-gray-500 uppercase">{state.history.length} BƯỚC</span>
            </div>
            {state.history.map((histState, idx) => (
              <button
                key={idx}
                onClick={() => setState(prev => ({
                  ...prev,
                  historyIndex: idx,
                  originalImage: prev.history[idx],
                  maskCanvas: null,
                  floatingObject: null,
                  status: `Đã tải bước ${idx + 1}`
                }))}
                className={`w-16 h-16 shrink-0 border-[3px] overflow-hidden relative ${state.historyIndex === idx ? 'border-black opacity-100 shadow-[2px_2px_0px_#000] -translate-y-1' : 'border-gray-300 opacity-60 hover:opacity-100'}`}
              >
                <img src={histState.dataUrl} className="w-full h-full object-cover" alt="History" />
                <div className="absolute top-0 left-0 bg-black text-white text-[9px] font-bold px-1 m-0.5">{idx + 1}</div>
              </button>
            ))}
          </div>
        )}
      </main>

      <Controls
        tool={state.tool}
        setTool={(tool) => setState(prev => ({ ...prev, tool }))}
        editMode={state.editMode}
        setEditMode={(editMode) => setState(prev => ({ ...prev, editMode, floatingObject: null, maskCanvas: null }))}
        moveSubMode={state.moveSubMode}
        setMoveSubMode={(moveSubMode) => setState(prev => ({ ...prev, moveSubMode }))}
        brushSize={state.brushSize}
        setBrushSize={(brushSize) => setState(prev => ({ ...prev, brushSize }))}
        modelVariant={state.modelVariant}
        setModelVariant={(modelVariant) => setState(prev => ({ ...prev, modelVariant }))}
        highQuality={state.highQuality}
        setHighQuality={(highQuality) => setState(prev => ({ ...prev, highQuality }))}
        onReset={handleResetMask}
        onSave={handleSave}
        onDone={handleDone}
        onUndo={handleUndo}
        onRedo={handleRedo}
        canUndo={state.historyIndex > 0}
        canRedo={state.historyIndex < state.history.length - 1}
        status={state.status}
        debugMode={state.debugMode}
        setDebugMode={(debugMode) => setState(prev => ({ ...prev, debugMode }))}
        hasFloatingObject={!!state.floatingObject}
        onCancelMove={() => setState(prev => ({ ...prev, floatingObject: null }))}
      />

      {state.debugSteps.length > 0 && (
        <DebugView steps={state.debugSteps} onClose={() => setState(prev => ({ ...prev, debugSteps: [] }))} />
      )}
    </div>
  );
}

import { FloatingObject } from '../types';

export const extractFloatingObject = (
  baseCanvas: HTMLCanvasElement,
  maskCanvas: HTMLCanvasElement
): FloatingObject | null => {
  const ctx = maskCanvas.getContext('2d')!;
  const imgData = ctx.getImageData(0, 0, maskCanvas.width, maskCanvas.height);
  const data = imgData.data;
  let minX = maskCanvas.width, minY = maskCanvas.height, maxX = 0, maxY = 0;
  let hasPixels = false;

  for (let y = 0; y < maskCanvas.height; y++) {
    for (let x = 0; x < maskCanvas.width; x++) {
      const alpha = data[(y * maskCanvas.width + x) * 4 + 3];
      if (alpha > 0) {
        if (x < minX) minX = x;
        if (x > maxX) maxX = x;
        if (y < minY) minY = y;
        if (y > maxY) maxY = y;
        hasPixels = true;
      }
    }
  }

  if (!hasPixels) return null;

  const padding = 10;
  minX = Math.max(0, minX - padding);
  minY = Math.max(0, minY - padding);
  maxX = Math.min(maskCanvas.width, maxX + padding);
  maxY = Math.min(maskCanvas.height, maxY + padding);

  const width = maxX - minX;
  const height = maxY - minY;

  const objCanvas = document.createElement('canvas');
  objCanvas.width = width;
  objCanvas.height = height;
  const objCtx = objCanvas.getContext('2d')!;

  objCtx.drawImage(baseCanvas, minX, minY, width, height, 0, 0, width, height);
  objCtx.globalCompositeOperation = 'destination-in';
  objCtx.drawImage(maskCanvas, minX, minY, width, height, 0, 0, width, height);

  const origMaskCanvas = document.createElement('canvas');
  origMaskCanvas.width = maskCanvas.width;
  origMaskCanvas.height = maskCanvas.height;
  origMaskCanvas.getContext('2d')!.drawImage(maskCanvas, 0, 0);

  return {
    canvas: objCanvas,
    originalMask: origMaskCanvas,
    x: minX,
    y: minY,
    width,
    height,
    scale: 1,
    rotation: 0
  };
};

const drawTransformed = (ctx: CanvasRenderingContext2D, floatingObj: FloatingObject, isMask = false) => {
  ctx.save();
  const cx = floatingObj.x + floatingObj.width / 2;
  const cy = floatingObj.y + floatingObj.height / 2;

  ctx.translate(cx, cy);
  ctx.rotate((floatingObj.rotation * Math.PI) / 180);
  ctx.scale(floatingObj.scale, floatingObj.scale);

  if (isMask) {
    const tempCanvas = document.createElement('canvas');
    tempCanvas.width = floatingObj.width;
    tempCanvas.height = floatingObj.height;
    const tCtx = tempCanvas.getContext('2d')!;
    tCtx.drawImage(floatingObj.canvas, 0, 0);
    tCtx.globalCompositeOperation = 'source-in';
    tCtx.fillStyle = 'white';
    tCtx.fillRect(0, 0, tempCanvas.width, tempCanvas.height);
    ctx.drawImage(tempCanvas, -floatingObj.width / 2, -floatingObj.height / 2);
  } else {
    ctx.drawImage(floatingObj.canvas, -floatingObj.width / 2, -floatingObj.height / 2);
  }
  ctx.restore();
};

export const createSimpleComposite = (
  originalImageCanvas: HTMLCanvasElement,
  floatingObj: FloatingObject
) => {
  const imgCanvas = document.createElement('canvas');
  imgCanvas.width = originalImageCanvas.width;
  imgCanvas.height = originalImageCanvas.height;
  const imgCtx = imgCanvas.getContext('2d')!;

  // Draw original image
  imgCtx.drawImage(originalImageCanvas, 0, 0);
  
  // Draw transformed object on top
  imgCtx.globalCompositeOperation = 'source-over';
  drawTransformed(imgCtx, floatingObj);

  const maskCanvas = document.createElement('canvas');
  maskCanvas.width = originalImageCanvas.width;
  maskCanvas.height = originalImageCanvas.height;
  const maskCtx = maskCanvas.getContext('2d')!;

  // Draw original mask (hole) in WHITE
  const tempHoleCanvas = document.createElement('canvas');
  tempHoleCanvas.width = maskCanvas.width;
  tempHoleCanvas.height = maskCanvas.height;
  const tempHoleCtx = tempHoleCanvas.getContext('2d')!;
  tempHoleCtx.drawImage(floatingObj.originalMask, 0, 0);
  tempHoleCtx.globalCompositeOperation = 'source-in';
  tempHoleCtx.fillStyle = '#FFFFFF'; // White
  tempHoleCtx.fillRect(0, 0, maskCanvas.width, maskCanvas.height);
  
  maskCtx.drawImage(tempHoleCanvas, 0, 0);

  return {
    imageBase64: imgCanvas.toDataURL('image/png').split(',')[1],
    maskBase64: maskCanvas.toDataURL('image/png').split(',')[1],
    maskCanvas
  };
};

export const createOneStepComposite = (
  originalImageCanvas: HTMLCanvasElement,
  floatingObj: FloatingObject
) => {
  const imgCanvas = document.createElement('canvas');
  imgCanvas.width = originalImageCanvas.width;
  imgCanvas.height = originalImageCanvas.height;
  const imgCtx = imgCanvas.getContext('2d')!;

  // Draw original image
  imgCtx.drawImage(originalImageCanvas, 0, 0);
  
  // Draw transformed object on top
  imgCtx.globalCompositeOperation = 'source-over';
  drawTransformed(imgCtx, floatingObj);

  const maskCanvas = document.createElement('canvas');
  maskCanvas.width = originalImageCanvas.width;
  maskCanvas.height = originalImageCanvas.height;
  const maskCtx = maskCanvas.getContext('2d')!;

  // Draw original mask (hole) in MAGENTA
  const tempHoleCanvas = document.createElement('canvas');
  tempHoleCanvas.width = maskCanvas.width;
  tempHoleCanvas.height = maskCanvas.height;
  const tempHoleCtx = tempHoleCanvas.getContext('2d')!;
  tempHoleCtx.drawImage(floatingObj.originalMask, 0, 0);
  tempHoleCtx.globalCompositeOperation = 'source-in';
  tempHoleCtx.fillStyle = '#FF00FF'; // Magenta
  tempHoleCtx.fillRect(0, 0, maskCanvas.width, maskCanvas.height);
  
  maskCtx.drawImage(tempHoleCanvas, 0, 0);

  // Draw transformed mask in GREEN
  const tempObjCanvas = document.createElement('canvas');
  tempObjCanvas.width = maskCanvas.width;
  tempObjCanvas.height = maskCanvas.height;
  const tempObjCtx = tempObjCanvas.getContext('2d')!;
  drawTransformed(tempObjCtx, floatingObj, true);
  tempObjCtx.globalCompositeOperation = 'source-in';
  tempObjCtx.fillStyle = '#00FF00'; // Green
  tempObjCtx.fillRect(0, 0, maskCanvas.width, maskCanvas.height);

  maskCtx.globalCompositeOperation = 'source-over';
  maskCtx.drawImage(tempObjCanvas, 0, 0);

  return {
    imageBase64: imgCanvas.toDataURL('image/png').split(',')[1],
    maskBase64: maskCanvas.toDataURL('image/png').split(',')[1],
    maskCanvas
  };
};

export const createTwoStepPhase1 = (
  originalImageCanvas: HTMLCanvasElement,

export const compositeInPlaceEdit = (
  originalImage: HTMLCanvasElement | HTMLImageElement,
  resultImage: HTMLImageElement,
  maskCanvas: HTMLCanvasElement
): string => {
  const canvas = document.createElement('canvas');
  canvas.width = originalImage.width;
  canvas.height = originalImage.height;
  const ctx = canvas.getContext('2d')!;

  // Draw original image
  ctx.drawImage(originalImage, 0, 0);

  // Create a temporary canvas for the masked result
  const tempCanvas = document.createElement('canvas');
  tempCanvas.width = canvas.width;
  tempCanvas.height = canvas.height;
  const tempCtx = tempCanvas.getContext('2d')!;

  // Draw the result image
  tempCtx.drawImage(resultImage, 0, 0, canvas.width, canvas.height);

  // Apply the mask (destination-in keeps the result only where the mask is opaque)
  tempCtx.globalCompositeOperation = 'destination-in';
  
  // Create a blurred version of the mask for smoother compositing
  const blurredMaskCanvas = document.createElement('canvas');
  blurredMaskCanvas.width = canvas.width;
  blurredMaskCanvas.height = canvas.height;
  const blurredMaskCtx = blurredMaskCanvas.getContext('2d')!;
  blurredMaskCtx.filter = 'blur(2px)';
  blurredMaskCtx.drawImage(maskCanvas, 0, 0);

  tempCtx.drawImage(blurredMaskCanvas, 0, 0, canvas.width, canvas.height);

  // Draw the masked result over the original image
  ctx.globalCompositeOperation = 'source-over';
  ctx.drawImage(tempCanvas, 0, 0);

  return canvas.toDataURL('image/png').split(',')[1];
};

import React, { useRef, useEffect, useState, useCallback } from 'react';
import { Tool, EditMode, FloatingObject } from '../types';
import { TransformNode } from './TransformNode';
import { extractFloatingObject } from '../utils/imageUtils';

interface EditorProps {
  image: string;
  tool: Tool;
  brushSize: number;
  onMaskUpdate: (canvas: HTMLCanvasElement | null) => void;
  status: string;
  setStatus: (s: string) => void;
  editMode: EditMode;
  floatingObject: FloatingObject | null;
  setFloatingObject: (obj: FloatingObject | null) => void;
  isGenerating?: boolean;
  onPromptRequest?: (pos: { x: number; y: number }) => void;
}

export const Editor: React.FC<EditorProps> = ({ 
  image, 
  tool, 
  brushSize, 
  onMaskUpdate,
  editMode,
  floatingObject,
  setFloatingObject,
  isGenerating,
  onPromptRequest
}) => {
  const containerRef = useRef<HTMLDivElement>(null);
  const baseCanvasRef = useRef<HTMLCanvasElement>(null);
  const maskCanvasRef = useRef<HTMLCanvasElement>(null);
  
  // Transform State
  const [transform, setTransform] = useState({ x: 0, y: 0, scale: 1 });
  const [isPanning, setIsPanning] = useState(false);
  const [isSpacePressed, setIsSpacePressed] = useState(false);
  
  // Interaction State
  const [isDrawing, setIsDrawing] = useState(false);
  const [mousePos, setMousePos] = useState({ x: 0, y: 0 });
  const [isHoveringImage, setIsHoveringImage] = useState(false);

  // Draw base image and hole
  useEffect(() => {
    if (!image || !baseCanvasRef.current) return;
    const img = new Image();
    img.src = image;
    img.onload = () => {
      const canvas = baseCanvasRef.current!;
      canvas.width = img.width;
      canvas.height = img.height;
      
      const mCanvas = maskCanvasRef.current!;
      mCanvas.width = img.width;
      mCanvas.height = img.height;

      const ctx = canvas.getContext('2d')!;
      ctx.drawImage(img, 0, 0);

      if (floatingObject) {
        ctx.globalCompositeOperation = 'destination-out';
        ctx.drawImage(floatingObject.originalMask, 0, 0);
        ctx.globalCompositeOperation = 'source-over';
      }
      
      resetViewport();
    };
    // eslint-disable-next-line react-hooks/exhaustive-deps
  }, [image, floatingObject?.originalMask]);

  // Attach non-passive wheel listener for zoom
  useEffect(() => {
    const container = containerRef.current;
    if (!container) return;

    const handleWheel = (e: WheelEvent) => {
      e.preventDefault();
      if (e.ctrlKey || e.metaKey) {
        const delta = -e.deltaY;
        const targetZoomFactor = Math.pow(1.1, delta / 100);
        
        setTransform(prev => {
          const newScale = Math.max(0.1, Math.min(10, prev.scale * targetZoomFactor));
          const actualZoom = newScale / prev.scale;
          
          const rect = container.getBoundingClientRect();
          const relX = e.clientX - rect.left - rect.width / 2;
          const relY = e.clientY - rect.top - rect.height / 2;
          
          const newX = relX - (relX - prev.x) * actualZoom;
          const newY = relY - (relY - prev.y) * actualZoom;
          
          return { x: newX, y: newY, scale: newScale };
        });
      } else {
        setTransform(prev => ({
          ...prev,
          x: prev.x - e.deltaX,
          y: prev.y - e.deltaY
        }));
      }
    };

    container.addEventListener('wheel', handleWheel, { passive: false });
    return () => container.removeEventListener('wheel', handleWheel);
  }, []);

  // Handle Global Keys
  useEffect(() => {
    const handleKeyDown = (e: KeyboardEvent) => {
      if (e.code === 'Space') setIsSpacePressed(true);
      if (e.code === 'Escape') {
        if (floatingObject) {
          setFloatingObject(null);
        }
        const mainCanvas = maskCanvasRef.current;
        if (mainCanvas) {
          const ctx = mainCanvas.getContext('2d');
          ctx?.clearRect(0, 0, mainCanvas.width, mainCanvas.height);
        }
        onMaskUpdate(null);
        setIsDrawing(false);
        setIsPanning(false);
      }
    };
    const handleKeyUp = (e: KeyboardEvent) => {
      if (e.code === 'Space') setIsSpacePressed(false);
    };
    window.addEventListener('keydown', handleKeyDown);
    window.addEventListener('keyup', handleKeyUp);
    return () => {
      window.removeEventListener('keydown', handleKeyDown);
      window.removeEventListener('keyup', handleKeyUp);
    };
  }, [floatingObject, setFloatingObject, onMaskUpdate]);

  // Clear mask when editMode changes
  useEffect(() => {
    const mainCanvas = maskCanvasRef.current;
    if (mainCanvas) {
      const ctx = mainCanvas.getContext('2d');
      ctx?.clearRect(0, 0, mainCanvas.width, mainCanvas.height);
    }
    onMaskUpdate(null);
    setIsDrawing(false);
    setIsPanning(false);
  }, [editMode, onMaskUpdate]);

  const getNormalizedCoords = (e: React.MouseEvent | MouseEvent): [number, number] | null => {
    const canvas = baseCanvasRef.current;
    if (!canvas) return null;
    const rect = canvas.getBoundingClientRect();
    const x = (e.clientX - rect.left) / rect.width;
    const y = (e.clientY - rect.top) / rect.height;
    if (x < 0 || x > 1 || y < 0 || y > 1) return null;
    return [x, y];
  };

  const handleMouseMove = (e: React.MouseEvent) => {
    setMousePos({ x: e.clientX, y: e.clientY });

    if (isPanning || (isSpacePressed && isDrawing)) {
      setTransform(prev => ({
        ...prev,
        x: prev.x + e.movementX,
        y: prev.y + e.movementY
      }));
      return;
    }

    const coords = getNormalizedCoords(e);
    if (!coords) {
      setIsHoveringImage(false);
      return;
    }

    setIsHoveringImage(true);

    if (tool === 'brush' && isDrawing && !floatingObject) {
      const canvas = maskCanvasRef.current;
      if (!canvas) return;
      const ctx = canvas.getContext('2d');
      if (!ctx) return;
      const pxX = coords[0] * canvas.width;
      const pxY = coords[1] * canvas.height;
      ctx.globalCompositeOperation = 'source-over';
      ctx.fillStyle = '#ffffff';
      ctx.beginPath();
      ctx.arc(pxX, pxY, brushSize, 0, Math.PI * 2);
      ctx.fill();
      onMaskUpdate(canvas);
    }
  };

  const handleMouseDown = (e: React.MouseEvent) => {
    if (e.button === 1 || (e.button === 0 && isSpacePressed)) {
      setIsPanning(true);
      return;
    }

    if (e.button !== 0) return;
    if (floatingObject) return;

    if (tool === 'brush') {
      setIsDrawing(true);
      handleMouseMove(e);
    }
  };

  const handleMouseUp = useCallback((e: MouseEvent | React.MouseEvent) => {
    if (isDrawing && tool === 'brush' && !floatingObject) {
      if (editMode === 'move') {
        const mainCanvas = maskCanvasRef.current;
        const baseCanvas = baseCanvasRef.current;
        if (mainCanvas && baseCanvas) {
          const obj = extractFloatingObject(baseCanvas, mainCanvas);
          if (obj) {
            setFloatingObject(obj);
            const ctx = mainCanvas.getContext('2d');
            ctx?.clearRect(0, 0, mainCanvas.width, mainCanvas.height);
            onMaskUpdate(mainCanvas);
            onPromptRequest?.({ x: e.clientX, y: e.clientY });
          }
        }
      } else {
        onPromptRequest?.({ x: e.clientX, y: e.clientY });
      }
    }
    setIsDrawing(false);
    setIsPanning(false);
  }, [isDrawing, tool, editMode, floatingObject, setFloatingObject, onMaskUpdate, onPromptRequest]);

  useEffect(() => {
    window.addEventListener('mouseup', handleMouseUp);
    return () => window.removeEventListener('mouseup', handleMouseUp);
  }, [handleMouseUp]);

  const resetViewport = () => {
    const container = containerRef.current;
    const canvas = baseCanvasRef.current;
    if (container && canvas && canvas.width > 0 && canvas.height > 0) {
      const containerRect = container.getBoundingClientRect();
      const scaleX = (containerRect.width * 0.9) / canvas.width;
      const scaleY = (containerRect.height * 0.9) / canvas.height;
      const scale = Math.min(scaleX, scaleY, 1);
      setTransform({ x: 0, y: 0, scale });
    } else {
      setTransform({ x: 0, y: 0, scale: 1 });
    }
  };

  return (
    <div 
      ref={containerRef}
      className={`relative w-full h-full bg-zinc-900 overflow-hidden flex items-center justify-center group select-none ${
        (isPanning || isSpacePressed) ? 'cursor-grab active:cursor-grabbing' : (floatingObject || isGenerating ? 'cursor-default' : 'cursor-none')
      }`}
      onMouseMove={handleMouseMove}
      onMouseDown={handleMouseDown}
      onMouseUp={handleMouseUp}
      onMouseLeave={() => { setIsHoveringImage(false); }}
      onContextMenu={(e) => e.preventDefault()}
    >
      <div 
        className="relative will-change-transform flex items-center justify-center"
        style={{ 
          transform: `translate(${transform.x}px, ${transform.y}px) scale(${transform.scale})`,
          transition: isPanning ? 'none' : 'transform 0.05s linear'
        }}
      >
        <div className="relative">
          <canvas 
            ref={baseCanvasRef} 
            className="pointer-events-none select-none bg-checkerboard block" 
          />
          <canvas 
            ref={maskCanvasRef} 
            className={`absolute inset-0 w-full h-full pointer-events-none transition-all duration-500 ${isGenerating ? 'animate-magical-wave' : ''}`} 
            style={{ display: floatingObject ? 'none' : 'block', opacity: isGenerating ? undefined : 0.65 }}
          />
          {floatingObject && (
            <TransformNode
              floatingObject={floatingObject}
              onChange={setFloatingObject}
              containerScale={transform.scale}
              isGenerating={isGenerating}
            />
          )}
        </div>
      </div>

      {!isPanning && !isSpacePressed && isHoveringImage && !floatingObject && !isGenerating && (
        <div 
          className="fixed pointer-events-none z-50 rounded-full border border-white mix-blend-difference transition-[opacity] duration-150"
          style={{
            left: mousePos.x,
            top: mousePos.y,
            width: (brushSize * (baseCanvasRef.current?.getBoundingClientRect().width || 1) / (maskCanvasRef.current?.width || 1)) * 2,
            height: (brushSize * (baseCanvasRef.current?.getBoundingClientRect().width || 1) / (maskCanvasRef.current?.width || 1)) * 2,
            transform: 'translate(-50%, -50%)',
            backgroundColor: 'rgba(255, 255, 255, 0.4)',
            boxShadow: '0 0 0 1px rgba(0,0,0,0.5)'
          }}
        />
      )}

      {/* Navigation & Controls Overlay */}
      <div className="absolute top-4 left-4 flex flex-col gap-3 pointer-events-none">
        <div className="bg-white text-black px-4 py-2 border-[3px] border-black text-xl font-bold uppercase tracking-widest shadow-[4px_4px_0px_#000]">
          THU PHÓNG {Math.round(transform.scale * 100)}%
        </div>
        <button 
          onClick={(e) => { e.stopPropagation(); resetViewport(); }}
          className="pointer-events-auto bg-[#e0e0e0] text-black hover:bg-black hover:text-white border-[3px] border-black px-4 py-2 text-xl font-bold uppercase tracking-widest transition-none shadow-[4px_4px_0px_#000]"
        >
          MẶC ĐỊNH
        </button>
      </div>

      <div className="absolute top-4 right-4 bg-white text-black px-4 py-2 border-[3px] border-black text-lg font-bold uppercase tracking-widest pointer-events-none shadow-[4px_4px_0px_#000] -translate-y-2 opacity-0 group-hover:opacity-100 group-hover:translate-y-0 transition-all duration-200">
        <span className="bg-black text-white px-2 py-1 mr-2">CỌ VẼ {brushSize}</span>
        SPACE: KÉO / CUỘN: ZOOM
      </div>
    </div>
  );
};

import React, { useState, useEffect } from 'react';
import { FloatingObject } from '../types';
import { RotateCw } from 'lucide-react';

interface Props {
  floatingObject: FloatingObject;
  onChange: (obj: FloatingObject) => void;
  containerScale: number;
  isGenerating?: boolean;
}

export const TransformNode: React.FC<Props> = ({ floatingObject, onChange, containerScale, isGenerating }) => {
  const [activeAction, setActiveAction] = useState<string | null>(null);
  const [startPos, setStartPos] = useState({ x: 0, y: 0 });
  const [startObj, setStartObj] = useState<FloatingObject | null>(null);

  useEffect(() => {
    if (!activeAction) return;

    const handlePointerMove = (e: PointerEvent) => {
      if (!startObj) return;
      const dx = (e.clientX - startPos.x) / containerScale;
      const dy = (e.clientY - startPos.y) / containerScale;

      let newObj = { ...startObj };

      if (activeAction === 'move') {
        newObj.x += dx;
        newObj.y += dy;
      } else if (activeAction === 'scale') {
        // Calculate new scale based on Euclidean distance from center vs original
        // Center in screen space when interaction started
        const centerX = startPos.x - dx; // Rough approximation, but we can just use movement magnitude
        // Simple scale logic: if you drag out from center it scales up
        const moveDist = dx + dy; // TopLeft -> BottomRight is positive dx, dy
        let scaleChange = moveDist / startObj.width;
        if (e.clientX < startPos.x && e.clientY < startPos.y) {
           scaleChange = -scaleChange; // Invert for top-left
        } else if (e.clientX > startPos.x && e.clientY < startPos.y) {
           scaleChange = (dx - dy) / startObj.width;
        } else if (e.clientX < startPos.x && e.clientY > startPos.y) {
           scaleChange = (-dx + dy) / startObj.width;
        }
        
        newObj.scale = Math.max(0.1, startObj.scale + scaleChange);
      } else if (activeAction === 'rotate') {
        // Simplified rotation based on horizontal drag
        newObj.rotation += dx;
      }

      onChange(newObj);
    };

    const handlePointerUp = () => {
      setActiveAction(null);
    };

    window.addEventListener('pointermove', handlePointerMove);
    window.addEventListener('pointerup', handlePointerUp);

    return () => {
      window.removeEventListener('pointermove', handlePointerMove);
      window.removeEventListener('pointerup', handlePointerUp);
    };
  }, [activeAction, startPos, startObj, containerScale, onChange]);

  const handlePointerDown = (e: React.PointerEvent, action: string) => {
    e.stopPropagation();
    setActiveAction(action);
    setStartPos({ x: e.clientX, y: e.clientY });
    setStartObj(floatingObject);
  };

  return (
    <div
      style={{
        position: 'absolute',
        left: floatingObject.x,
        top: floatingObject.y,
        width: floatingObject.width,
        height: floatingObject.height,
        transform: `rotate(${floatingObject.rotation}deg) scale(${floatingObject.scale})`,
        transformOrigin: 'center center',
        cursor: activeAction === 'move' ? 'grabbing' : 'grab',
        border: isGenerating ? 'none' : '2px dashed rgba(255, 255, 255, 0.8)',
        boxShadow: isGenerating ? 'none' : '0 0 0 1px rgba(0, 0, 0, 0.5)',
        touchAction: 'none'
      }}
      onPointerDown={(e) => !isGenerating && handlePointerDown(e, 'move')}
    >
      <img 
        src={floatingObject.canvas.toDataURL()} 
        className={isGenerating ? "animate-google-glow" : ""}
        style={{ width: '100%', height: '100%', pointerEvents: 'none' }} 
        alt="Extracted"
      />

      {!isGenerating && (
        <>
          {/* Handle styles */}
          {[
            { top: -6, left: -6, cursor: 'nwse-resize' },
            { top: -6, right: -6, cursor: 'nesw-resize' },
            { bottom: -6, left: -6, cursor: 'nesw-resize' },
            { bottom: -6, right: -6, cursor: 'nwse-resize' }
          ].map((pos, i) => (
            <div
              key={i}
              style={{
                position: 'absolute',
                ...pos,
                width: 12,
                height: 12,
                backgroundColor: 'white',
                border: '1px solid black',
                cursor: pos.cursor,
              }}
              onPointerDown={(e) => handlePointerDown(e, 'scale')}
            />
          ))}

          {/* Rotate Handle (Top Center) */}
          <div
            style={{
              position: 'absolute',
              top: -30,
              left: '50%',
              transform: 'translateX(-50%)',
              width: 24,
              height: 24,
              backgroundColor: 'white',
              border: '1px solid black',
              cursor: 'ew-resize',
              borderRadius: '50%',
              display: 'flex',
              alignItems: 'center',
              justifyContent: 'center',
              color: 'black'
            }}
            onPointerDown={(e) => handlePointerDown(e, 'rotate')}
          >
            <RotateCw size={14} />
          </div>
          
          {/* Connecting line for rotate handle */}
          <div style={{
            position: 'absolute',
            top: -18,
            left: '50%',
            width: 1,
            height: 18,
            backgroundColor: 'rgba(255,255,255,0.8)',
            borderLeft: '1px solid rgba(0,0,0,0.5)'
          }} />
        </>
      )}
    </div>
  );
};
